579 void findPU2SitesInDistance()
580 {
581 PU2SitesInDisplacementThreshold.clear();
582
583 int PUsNum = PUsToLegalize.size();
584
585 for (int i = 0; i < PUsNum; i++)
586 {
587 PU2SitesInDisplacementThreshold[PUsToLegalize[i]] = nullptr;
588 }
589
590#pragma omp parallel for
591 for (int i = 0; i < PUsNum; i++)
592 {
593 PlacementInfo::PlacementUnit *curPU = PUsToLegalize[i];
594 DesignInfo::DesignCell *curCell = nullptr;
595 if (auto unpackedCell = dynamic_cast<PlacementInfo::PlacementUnpackedCell *>(curPU))
596 {
597 curCell = unpackedCell->getCell();
598 }
599 else if (auto curMacro = dynamic_cast<PlacementInfo::PlacementMacro *>(curPU))
600 {
601 assert(curMacro->getCells().size() > 0);
602 curCell = curMacro->getCells()[0];
603 }
604 assert(curCell);
605 PU2SitesInDisplacementThreshold[curPU] = placementInfo->findNeiborSiteFromBinGrid(
606 curCell, cellLoc[curCell->getCellId()].X, cellLoc[curCell->getCellId()].Y, displacementThreshold,
607 candidateFactor * maxNumCandidate);
608 }

661 inline float getHPWLChange(PlacementInfo::PlacementUnit *curPU, DeviceInfo::DeviceSite *curSite)
662 {
663 assert(PU2Site2HPWLIncrease.find(curPU) != PU2Site2HPWLIncrease.end());
664 if (PU2Site2HPWLIncrease[curPU].find(curSite) != PU2Site2HPWLIncrease[curPU].end())
665 {
666 return PU2Site2HPWLIncrease[curPU][curSite];
667 }
668 float oriHPWL = 0.0;
669 float newHPWL = 0.0;
670 float PUX = curSite->X();
671 float PUY = curSite->Y();
672 auto nets = placementInfo->getPlacementUnitId2Nets()[curPU->getId()];
673
674 for (auto curNet : nets)
675 {
676 if (curNet->getDesignNet()->getPins().size() > 1000) // it could be clock
677 continue;
678 oriHPWL += curNet->getHPWL(y2xRatio);
679 newHPWL += curNet->getNewHPWLByTrying(curPU, PUX, PUY, y2xRatio);
680 }
681
682 PU2Site2HPWLIncrease[curPU][curSite] = (newHPWL - oriHPWL);
683 return (newHPWL - oriHPWL);
684 }

719 inline int sortPartition(PlacementInfo::PlacementUnit *curPU, std::vector<DeviceInfo::DeviceSite *> &sites, int low,
720 int high)
721 {
722 int pivot, index, i;
723 index = low;
724 pivot = high;
725 for (i = low; i < high; i++)
726 {
727 // finding index of pivot.
728 // if (a[i] < a[pivot])
729 // if (getDisplacement(PULoc, sites[i]) < getDisplacement(PULoc, sites[pivot]))
730 if (getHPWLChange(curPU, sites[i]) < getHPWLChange(curPU, sites[pivot]))
731 {
732 swapSitePtr(&sites[i], &sites[index]);
733 index++;
734 }
735 }
736 swapSitePtr(&sites[pivot], &sites[index]);
737 return index;

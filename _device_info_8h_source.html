656 inline void addClockAndCell(int clockNetId, int cellId, float x, float y)
657 {
658 float offsetXInRegion = x - leftX;
659 float offsetYInRegion = y - bottomY;
660
661 // this approach is not accurate enought since the slice columns are not evenly
662 // distributed in a clock region. However, it is fast for frequently checking of
663 // clock utilization
664 int colX = offsetXInRegion / colWidth;
665 int colY = offsetYInRegion / colHeight;
666 assert(colX >= -1);
667 assert(colY >= -1);
668 assert(clockColumns.size());
669 assert(colY <= (int)clockColumns.size());
670 assert(colX <= (int)clockColumns[0].size());
671 if (colX < 0)
672 colX = 0;
673 if (colY < 0)
674 colY = 0;
675 if (colX >= (int)clockColumns[0].size())
676 {
677 colX = clockColumns[0].size() - 1;
678 }
679 if (colY >= (int)clockColumns.size())
680 {
681 colY = clockColumns.size() - 1;
682 }
683 clockColumns[colY][colX].addClockNetId(clockNetId, cellId);

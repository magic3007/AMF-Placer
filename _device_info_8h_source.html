667 inline void addClockAndCell(int clockNetId, int cellId, float x, float y)
668 {
669 float offsetXInRegion = x - leftX;
670 float offsetYInRegion = y - bottomY;
671
672 // this approach is not accurate enought since the slice columns are not evenly
673 // distributed in a clock region. However, it is fast for frequently checking of
674 // clock utilization
675 int colX = offsetXInRegion / colWidth;
676 int colY = offsetYInRegion / colHeight;
677 assert(colX >= -1);
678 assert(colY >= -1);
679 assert(clockColumns.size());
680 assert(colY <= (int)clockColumns.size());
681 assert(colX <= (int)clockColumns[0].size());
682 if (colX < 0)
683 colX = 0;
684 if (colY < 0)
685 colY = 0;
686 if (colX >= (int)clockColumns[0].size())
687 {
688 colX = clockColumns[0].size() - 1;
689 }
690 if (colY >= (int)clockColumns.size())
691 {
692 colY = clockColumns.size() - 1;
693 }
694 clockColumns[colY][colX].addClockNetId(clockNetId, cellId);
695 }

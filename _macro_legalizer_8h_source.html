698 {
699 float oriHPWL = 0.0;
700 float newHPWL = 0.0;
701 auto tmpPU = placementInfo->getPlacementUnitByCell(curCell);
702 float PUX = 0.0, PUY = 0.0;
703 auto nets = placementInfo->getPlacementUnitId2Nets()[tmpPU->getId()];
704 float numCellsInMacro = 1.0;
705 if (dynamic_cast<PlacementInfo::PlacementUnpackedCell *>(tmpPU))
706 {
707 PUX = curSite->X();
708 PUY = curSite->Y();
709 }
710 else if (PlacementInfo::PlacementMacro *tmpMacro = dynamic_cast<PlacementInfo::PlacementMacro *>(tmpPU))

712
713 PUX = curSite->X() - tmpMacro->getCellOffsetXInMacro(curCell);
714 PUY = curSite->Y() - tmpMacro->getCellOffsetYInMacro(curCell);
715 numCellsInMacro = tmpMacro->getCells().size();
716 }

718 for (auto curNet : nets)
719 {
720 if (curNet->getDesignNet()->getPins().size() > 1000) // it could be clock
721 continue;
722 oriHPWL += curNet->getHPWL(y2xRatio);
723 newHPWL += curNet->getNewHPWLByTrying(tmpPU, PUX, PUY, y2xRatio);
724 }

791 inline int sortPartition(DesignInfo::DesignCell *curCell, std::vector<DeviceInfo::DeviceSite *> &sites, int low,
792 int high, PlacementInfo::Location &macroLoc)
793 {
794 int pivot, index, i;
795 index = low;
796 pivot = high;
797 for (i = low; i < high; i++)
798 {
799 // finding index of pivot.
800 // if (a[i] < a[pivot])
801 // if (getDisplacement(macroLoc, sites[i]) < getDisplacement(macroLoc, sites[pivot]))
802 if (getHPWLChange(curCell, sites[i]) < getHPWLChange(curCell, sites[pivot]))
803 {
804 swapSitePtr(&sites[i], &sites[index]);
805 index++;
806 }
807 }
808 swapSitePtr(&sites[pivot], &sites[index]);
809 return index;
810 }
811 inline int RandomPivotPartition(DesignInfo::DesignCell *curCell, std::vector<DeviceInfo::DeviceSite *> &sites,
812 int low, int high, PlacementInfo::Location &macroLoc)
813 {
814 // Random selection of pivot.
815 int pvt;
816 pvt = (high + low) / 2;
817 // pvt = low + n % (high - low + 1); // Randomizing the pivot value from sub-array.
818 swapSitePtr(&sites[high], &sites[pvt]);
819 return sortPartition(curCell, sites, low, high, macroLoc);
820 }

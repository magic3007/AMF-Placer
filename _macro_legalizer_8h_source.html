372 float PUX = 0.0, PUY = 0.0;
373 auto nets = placementInfo->getPlacementUnitId2Nets()[tmpPU->getId()];

376 {

452 inline int sortPartition(DesignInfo::DesignCell *curCell, std::vector<DeviceInfo::DeviceSite *> &sites, int low,
453 int high, PlacementInfo::Location &macroLoc)
454 {
455 int pivot, index, i;
456 index = low;
457 pivot = high;
458 for (i = low; i < high; i++)
459 {
460 // finding index of pivot.
461 // if (a[i] < a[pivot])
462 // if (getDisplacement(macroLoc, sites[i]) < getDisplacement(macroLoc, sites[pivot]))
463 if (getHPWLChange(curCell, sites[i]) < getHPWLChange(curCell, sites[pivot]))
464 {
465 swapSitePtr(&sites[i], &sites[index]);
466 index++;
467 }
468 }
469 swapSitePtr(&sites[pivot], &sites[index]);
470 return index;
471 }
472 inline int RandomPivotPartition(DesignInfo::DesignCell *curCell, std::vector<DeviceInfo::DeviceSite *> &sites,
473 int low, int high, PlacementInfo::Location &macroLoc)
474 {
475 // Random selection of pivot.
476 int pvt;
477 pvt = (high + low) / 2;
478 // pvt = low + n % (high - low + 1); // Randomizing the pivot value from sub-array.
479 swapSitePtr(&sites[high], &sites[pvt]);
480 return sortPartition(curCell, sites, low, high, macroLoc);
481 }

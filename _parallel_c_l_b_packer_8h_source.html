301 inline unsigned int getPairPinNum(DesignInfo::DesignCell *LUTA, DesignInfo::DesignCell *LUTB)
302 {
303 if (LUTA->getInputPins().size() == 6 || LUTB->getInputPins().size() == 6 || LUTA->isLUT6() ||
304 LUTB->isLUT6())
305 return 12;
306
307 int pinNumA = 0;
308 int totalPin = 0;
309 int netIds[5]; // be aware that a LUT might have pins connected to the same net and they should be
310 // treated as different inputs.
311
312 for (auto tmpPin : LUTA->getInputPins())
313 {
314 if (!tmpPin->isUnconnected())
315 {
316 netIds[pinNumA] = tmpPin->getNet()->getElementIdInType();
317 pinNumA++;
318 }
319 }
320 totalPin = pinNumA;
321 for (auto tmpPin : LUTB->getInputPins())
322 {
323 if (!tmpPin->isUnconnected())
324 {
325 bool matched = false;
326 for (int i = 0; i < pinNumA; i++)
327 {
328 if (netIds[i] >= 0 && netIds[i] == tmpPin->getNet()->getElementIdInType())
329 {
330 netIds[i] = -1;
331 matched = true;
332 break;
333 }
334 }
335 if (!matched)
336 {
337 totalPin++;
338 }
339 }
340 }
341

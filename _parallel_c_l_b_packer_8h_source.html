312 inline unsigned int getPairPinNum(DesignInfo::DesignCell *LUTA, DesignInfo::DesignCell *LUTB)
313 {
314 if (LUTA->getInputPins().size() == 6 || LUTB->getInputPins().size() == 6 || LUTA->isLUT6() ||
315 LUTB->isLUT6())
316 return 12;
317
318 int pinNumA = 0;
319 int totalPin = 0;
320 int netIds[5]; // be aware that a LUT might have pins connected to the same net and they should be
321 // treated as different inputs.
322
323 for (auto tmpPin : LUTA->getInputPins())
324 {
325 if (!tmpPin->isUnconnected())
326 {
327 netIds[pinNumA] = tmpPin->getNet()->getElementIdInType();
328 pinNumA++;
329 }
330 }
331 totalPin = pinNumA;
332 for (auto tmpPin : LUTB->getInputPins())
333 {
334 if (!tmpPin->isUnconnected())
335 {
336 bool matched = false;
337 for (int i = 0; i < pinNumA; i++)
338 {
339 if (netIds[i] >= 0 && netIds[i] == tmpPin->getNet()->getElementIdInType())
340 {
341 netIds[i] = -1;
342 matched = true;
343 break;
344 }
345 }
346 if (!matched)
347 {
348 totalPin++;
349 }
350 }
351 }
352
353 return totalPin;
